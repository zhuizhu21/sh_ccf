#include <iostream>
#include <string>
#include <stack>

using namespace std;

bool isPrior(char a,char b){
    if((a=='x'||a=='/')&&(b=='+'||b=='-')){
        return true;
    }else {
        return false;
    }
}

string doChange(string s){
    stack<char> s2;
    string s1="";
    for(int i=0;i<int(s.length());i++){
        if(s[i]>='0'&&s[i]<='9'){
            s1+=s[i];
        }else {
            if(s2.empty()){
                s2.push(s[i]);
            }else {
                while ((!s2.empty())&&(!isPrior(s[i],s2.top()))) {
                    s1+=s2.top();
                    s2.pop();
                }
                s2.push(s[i]);
            }
        }
    }
    while(!s2.empty()){
        s1+=s2.top();
        s2.pop();
    }
    return s1;
}

void doCalculate(string s){
    int res=0;
    stack<int> s1;
    for(int i=0;i<int(s.length());i++){
        if(s[i]>='0'&&s[i]<='9'){
            s1.push(s[i]-'0');
        }else{
            int b=s1.top();
            s1.pop();
            int a=s1.top();
            s1.pop();
//            cout<<a<<s[i]<<b<<endl;
            switch (s[i]) {
            case '+':
                res=a+b;
                s1.push(res);
                break;
            case '-':
                res=a-b;
                s1.push(res);
                break;
            case 'x':
                res=a*b;
                s1.push(res);
                break;
            case '/':
                res=a/b;
                s1.push(res);
                break;
            default:
                break;
            }
        }
    }
//    cout<<res<<endl;
    if(res==24){
        cout<<"Yes"<<endl;
    }else {
        cout<<"No"<<endl;
    }
}

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        string s;
        string res;
        cin>>s;
        res=doChange(s);
        doCalculate(res);
    }

    return 0;
}```
